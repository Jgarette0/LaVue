


  #Steps run this project into your local (ithink)


1.
        
                git clone https://github.com/Jgarette0/LaVue.git

        
2.
        
                cd lavue
        
3.
        
              composer install
        
3.
        
                npm install
        
4.
        
                cp .env.example .env
        
5.
        
                php artisan key:generate

6.
        
                 npm run dev
        
7.
        
                 php artisan serve
